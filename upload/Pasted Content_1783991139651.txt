                function runExport(btnId) {
                    var btn = document.getElementById(btnId); if (!btn || btn.classList.contains('loading')) return;
                    btn.classList.add('loading'); btn.disabled = true;
                    if (typeof JSZip === 'undefined') { btn.classList.remove('loading'); btn.disabled = false; showToast('error', 'Export failed', 'JSZip could not be loaded.', 5000); return }

                    setTimeout(function () { showToast('info', 'Generating files', 'notaflow-tokens.json, theme.css, fonts.css...', 3500) }, 900);
                    setTimeout(function () {
                        try {
                            var zip = new JSZip();
                            zip.file('notaflow-tokens.json', JSON.stringify(buildTokenJSON(), null, 2));
                            zip.file('theme.css', buildThemeCSS());
                            zip.file('fonts.css', buildFontsCSS());
                            zip.generateAsync({ type: 'blob' }).then(function (blob) {

                            }).catch(function (e) { btn.classList.remove('loading'); btn.disabled = false; showToast('error', 'Export failed', 'ZIP error: ' + (e.message || 'Unknown'), 5000) });
